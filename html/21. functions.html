function addNumber (x, y){
    return x + y
}
function substractNumber(x, y){
    return x - y
}
function multiply (x, y){
    return x * y
}
function divide (x, y){
    return x / y
}
function evenNumber(number){
    if (number % 2 === 0){
        return true
    }
    else{
        return false
    }
}
function isvalidEmail(email){
    return email.includes('@') ? true:false
    // ternary operator
}
